    function mygcd(x,y){
        let div = Math.max(x,y) % Math.min(x,y); 
        if (div != 0) { 
            let result = Math.min(x,y) % div;
            x = Math.min(x,y);
            y = div; 
            return (result == 0) ? div : mygcd(x,y); 
        } else return Math.min (x,y); 
    }
